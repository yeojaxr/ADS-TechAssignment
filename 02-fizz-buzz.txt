BEGIN
    SET counter = 0
    SET n = input number

    WHILE counter < n
		INCREMENT counter
        IF n%3 = 0
            DISPLAY "Fizz"
        IF n%5 = 0
            DISPLAY "Buzz"
        IF n%3=0 AND n%5=0
            DISPLAY "FizzBuzz"
        ELSE
            DISPLAY n
        ENDIF
    ENDWHILE
END
